    \question[4]
    Resuelve los siguientes sistemas de ecuaciones:
    \begin{parts}
        \part
        Por el método de sustitución:
        \begin{flushleft}
            $\begin{cases}
                    \nonumber
                    2x - 3y  = -4 \\
                    \nonumber
                    3x +2y = 7
                \end{cases}$
        \end{flushleft}
        \vspace{\stretch{1}}

        \part
        Por el método de igualación.
        \begin{flushleft}
            $\begin{cases}
                    \nonumber
                    2x + y = 10 \\
                    \nonumber
                    -3x - 2y = -16
                \end{cases}$
        \end{flushleft}
        \vspace{\stretch{1}}

        \part
        Por el método de reducción.
        \begin{flushleft}
            $\begin{cases}
                    \nonumber
                    x + 2y = 6 \\
                    \nonumber
                    x + 3y = 7
                \end{cases}$
        \end{flushleft}
        \vspace{\stretch{1}}
    \end{parts}

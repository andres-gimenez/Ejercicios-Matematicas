        \question
        Resuelve las siguiente ecuaciones de primer grado.
        \begin{parts}
            \part
                $3x+1 = 2$
            \part 
                $2x = 3-3x$
            \part 
                $\frac{2}{3}x - 4 = \frac{4x}{2}$
            \part
                $4(x-3)-2(3x-4) = 0$
            \part
                $5x-(3-x) = 2(4x-1)$
            \part
                $x+2(x+1) = 2(4x-1)$
            \part
                $2-(3-4x) = 5-(6-7x)$
            \part
                $3x+3(2x+2) = 5x$
            \part
                $3x + 2 = 3x$
            \part
                $\frac{2x}{2} = x$
        \end{parts}

        \question
        Resuelve las siguiente ecuaciones de segundo grado.
        \begin{parts}
            \part
                $x^2-7x+6 = 0$
                \begin{solution}
                    $\begin{array}{ccc} 
                        & & x_1 = \frac{7+5}{2}=1\\ 
                        & \nearrow &\\ 
                        x=\frac{7\pm \sqrt{7^2 - 4 \cdot 1 \cdot 6}}{2 \cdot1}= \frac{7\pm \sqrt{25}}{2}
                        & &\\ 
                        & \searrow &\\
                        & &x_2 = \frac{7-5}{2}=6
                    \end{array}$
                \end{solution}
            \part
                $3x^2-27 = 0$
            \part
                $2x^2-8x = 0$
            \part
                $x^2-6x+9 = 0$
            \part
                $x^2-3x+8 = 0$
            \part 
                $-2x^2+26x-60 = 0$
        \end{parts}
                
        \question
        Resuelve las siguiente ecuaciones de segundo grado expresandolas primero en su forma canonica
        ($ax^2+bx+c = 0$).
        \begin{parts}
            \part
                $(x-1)(x+2) = 28$
            \part 
                $2x^2-(x+1)(x+3) = 27$
            \part
                $\frac{\left(x^2-2x\right)}{3} = 3$
        \end{parts}
                
        \question
        Resuelve las ecuaciones
        \begin{parts}
            \part
                $(x+7)(x-6) = 0$
            \part
                $(x+14)(2x-8) = 0$
            \part
                $x(x+6) = 0$
            \part
                $5(3x-3)(x-1) = 0$
            \part
                $(x+2)^2 = 0$
        \end{parts}
    \end{questions}

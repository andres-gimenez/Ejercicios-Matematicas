        Resuelve los siguiens sistemas de ecuaciones por el método gráfico.
        \begin{parts}
            \part
            \begin{flushleft}
                $\begin{cases}
                    \nonumber
                    2x + y  = 2 \\
                    \nonumber
                    x + y = 4 
                \end{cases}$
            \end{flushleft}

            \begin{solution}

                \begin{multicols}{2}
                    \center
                    $2x + y = 2$ 
                    \vspace{ 1mm } \\
                    \begin{tabular}{c|c} 
                        \textbf{x} & \textbf{y} \\
                        \hline
                        1 & 0 \\
                        0 & 2 \\
                    \end{tabular}

                    \center
                    $ x + y = 4$ 
                    \vspace{ 1mm } \\
                    \begin{tabular}{c|c} 
                        \textbf{x} & \textbf{y} \\
                        \hline
                        0 & 4 \\
                        4 & 0 \\
                    \end{tabular}
                \end{multicols}

                \begin{tikzpicture}[scale=0.7]
                    \tkzInit[xmax=8,ymax=8,xmin=-8,ymin=-8]

        \question
        Divide los siguientes monomios
        \begin{parts}
            \part
            $\polynomialfrac[reciprocal]{-24,0,0,0}{3,0,0}$
            \begin{solution}
                $\polynomialfrac[reciprocal]{-24,0,0,0}{3,0,0} = \boxed{-8x}$
            \end{solution}

            \part
            $\frac{-45x^2t^4}{10x^2t^3}$
            \begin{solution}
                $\frac{-45x^2t^4}{10x^2t^3} = -\frac{9t}{2} = \boxed{-\frac{9}{2}t}$
            \end{solution}

            \part
            $\frac{8x^2}{16xy}$
            \begin{solution}
                $\frac{8x^2}{16xy} = \boxed{\frac{x}{2y} = \frac{1}{2} \frac{x}{y}}$
            \end{solution}
        \end{parts}

        \question
        Realiza las siguientes divisiones de polinomios

        \begin{parts}
            \part
            $\left(\polynomial[reciprocal]{3,-4,5}\right) : \left( \polynomial[reciprocal]{1,-1} \right)$

            \begin{solution}
                \nopagebreak
                $$\polylongdiv[style=D]{3x^2-4x+5}{x-1}$$
                $$\boxed{\polynomial[reciprocal]{3,-4,5} = \left( \polynomial[reciprocal]{1,-1} \right) \left( \polynomial[reciprocal]{3,-1} \right) + 4}$$
            \end{solution}

            \newpage
            \part
            $\polynomialfrac[reciprocal]{6,-1,-25,15}{2,3,-4}$
            \begin{solution}
                $$\polylongdiv[style=D]{6x^3-x^2-25x+15}{2x^2+3x-4}$$
                $$\boxed{\polynomial[reciprocal]{6,-1,-25,15} = \left( \polynomial[reciprocal]{2,3,-4} \right) \left( \polynomial[reciprocal]{3,-5} \right) + \left( \polynomial[reciprocal]{2,-5} \right)}$$
            \end{solution}

            \part
            $\polynomialfrac[reciprocal]{1,5,-2,17,-18}{1,0,4}$
            \begin{solution}
                $$\polylongdiv[style=D]{x^4+5x^3-2x^2+17x-18}{x^2-4}$$
                $$\boxed{\polynomial[reciprocal]{1,5,-2,17,-18} = \left( \polynomial[reciprocal]{1,0,-4} \right) \left( \polynomial[reciprocal]{1,5, 2} \right) + \left( \polynomial[reciprocal]{37,-10} \right)}$$
            \end{solution}

            \part
            $\polynomialfrac[reciprocal]{1,-3,0,0}{1,0,-4}$
            \begin{solution}
                $$\polylongdiv[style=D]{x^4-3x^3}{x^2-4}$$
                $$\boxed{\polynomial[reciprocal]{1,-3,0,0,0} = \left( \polynomial[reciprocal]{1,0,-4} \right) \left( \polynomial[reciprocal]{1,-3,4} \right) + \left( \polynomial[reciprocal]{-12, 16} \right)}$$
            \end{solution}
        \end{parts}

        \newpage
        \question
        Realiza las siguientes divisiones de polinomios por el método de Ruffini
        \begin{parts}
            \part
            $\left(\polynomial[reciprocal]{3,-4,5}\right) : \left( \polynomial[reciprocal]{1,-1} \right)$
            \begin{solution}
                $$\polyhornerscheme[x=1,tutor=true,resultbottomrule=true,resultleftrule=true,tutorlimit=8,stage=8]{3x^2-4x+5}$$

                $$\boxed{\polynomial[reciprocal]{3,-4,5} = \left( \polynomial[reciprocal]{1,-1} \right) \left( \polynomial[reciprocal]{3,-1} \right) + 4}$$
            \end{solution}

            \part
            $\left(\polynomial[reciprocal]{2,-3,2,-3}\right) : \left( \polynomial[reciprocal]{1,2} \right)$
            \begin{solution}
                $$\polyhornerscheme[x=-2,tutor=true,resultbottomrule=true,resultleftrule=true,tutorlimit=8,stage=8]{2x^3-3x^2+2x-3}$$

                $$\boxed{\polynomial[reciprocal]{2,-3,2,-3} = \left( \polynomial[reciprocal]{1,2} \right) \left( \polynomial[reciprocal]{2,-7,16} \right) -35}$$
            \end{solution}

            \part
            $\left(\polynomial[reciprocal]{1,0,-1,0,2,0,0}\right) : \left( \polynomial[reciprocal]{1,-1} \right)$
            \begin{solution}
                $$\polyhornerscheme[x=1, resultbottomrule=true,resultleftrule=true]{x^6-x^4+2x^2}$$

                $$\boxed{\polynomial[reciprocal]{1,0,-1,0,2,0,0} = \left( \polynomial[reciprocal]{1,-1} \right) \left( \polynomial[reciprocal]{1,1,0,0,2,2} \right) +2}$$
            \end{solution}

            \part
            $\polynomialfrac[reciprocal]{1,3,-5,1,0,-1}{1,1}$
            \begin{solution}
                $$\polyhornerscheme[x=-1,resultbottomrule=true,resultleftrule=true]{x^5+3x^4-5x^3+x^2-1}$$

                $$\boxed{\polynomial[reciprocal]{1,3,-5,1,0,-1} = \left( \polynomial[reciprocal]{1,-1} \right) \left( \polynomial[reciprocal]{1,2,-7,8,-8} \right) + 7}$$
            \end{solution}
        \end{parts}

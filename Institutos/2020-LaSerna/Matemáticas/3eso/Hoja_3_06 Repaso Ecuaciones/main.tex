        Resuelve las siguientes ecuaciones de 2º grado.
        \begin{parts}
            \part
            $x^2-5x+6=0$
            \part
            $x^2+x-6=0$
            \part
            $2x^2-7x+3=0$
        \end{parts}

        \question
        Resuelve sin utilizar la formula de ecuaciones de 2º grado.
        \begin{parts}
            \part
            $(x+3)(x-5)=0$
            \begin{sample} Si tenemos la ecuación factorizada la pondemos resolver facilmente. \\
                $\begin{array}{ccc} 
                    & & x+3 = 0 \Rightarrow \boxed{x= -3}\\ 
                    & \nearrow &\\ 
                    (x+3)(x-5)=0
                    & &\\ 
                    & \searrow &\\
                    & & x-5 = 0 \Rightarrow \boxed{x= 5}\\ 
                \end{array}$
             \end{sample}
            \part
            $(x+3)(x-5)=0$
            \part
            $(x+4)^2=0$
            \part
            $(x-5)^5-9=0$
            \part
            $x^2=10x$
            \part
            $2x^2-18=0$
            \part
            $16x+4x^2=0$
        \end{parts}
    \end{questions}

        \question[2]

            $\left(\polynomial[reciprocal]{2,9,16,26}\right) : \left( \polynomial[reciprocal]{2,3,7} \right)$
            \ifprintanswers
                \begin{solution}
                    \nopagebreak
                    $$\polylongdiv[style=D]{2x^3+9x^2+16x+26}{2x^2+3x+7}$$
                    $$\boxed{\polynomial[reciprocal]{2,9,16,26} = \left( \polynomial[reciprocal]{2,3,7} \right) \left( \polynomial[reciprocal]{1,3} \right) + 5}$$

            $\left(\polynomial[reciprocal]{4,-8,-9,7}\right) : \left( \polynomial[reciprocal]{1, -3} \right)$
            
            \ifprintanswers
                \begin{solution}
                    $$\polyhornerscheme[x=3,resultbottomrule=true,resultleftrule=true]{4x^3-8x^2-9x+7}$$

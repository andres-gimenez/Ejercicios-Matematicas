    \question[3]
    Resuelve los siguientes sistemas de ecuaciones:
    \begin{parts}
        \part
        Por el método de sustitución:
        \begin{flushleft}
            $\begin{cases}
                    \nonumber
                    2x - 3y  = -1 \\
                    \nonumber
                    3x +2y = 5
                \end{cases}$
        \end{flushleft}
        \vspace{\stretch{1}}

        \part
        Por el método de igualación.
        \begin{flushleft}
            $\begin{cases}
                    \nonumber
                    -3x - 4y = 5 \\
                    \nonumber
                    -2x + 3y = 9
                \end{cases}$
        \end{flushleft}
        \vspace{\stretch{1}}

        \part
        Por el método de reducción.
        \begin{flushleft}
            $\begin{cases}
                    \nonumber
                    x + 2y = 5 \\
                    \nonumber
                    x + 3y = 6
                \end{cases}$
        \end{flushleft}
        \vspace{\stretch{1}}
    \end{parts}

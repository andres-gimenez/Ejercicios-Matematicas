\section{Logaritmos}

\begin{definition}[Logaritmo]
    Logaritmo en base $a$ de un número $x$ es el exponente al que hay que elevar la base $a$ para obtener el número $x$, o sea:
    \\
    \center
    $\log_a N =x \Leftrightarrow a^x=N$
\end{definition}
